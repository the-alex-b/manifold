34template <typename T>
35class Vec;
36
37/*
38 * Specialized vector implementation with multithreaded fill and uninitialized
39 * memory optimizations.
40 * Note that the constructor and resize function will not perform initialization
41 * if the parameter val is not set. Also, this implementation is a toy
42 * implementation that did not consider things like non-trivial
43 * constructor/destructor, please keep T trivial.
44 */
45template <typename T>
46class Vec : public VecView<T> {
47 public:
48 Vec() {}
49
50 // Note that the vector constructed with this constructor will contain
51 // uninitialized memory. Please specify `val` if you need to make sure that
52 // the data is initialized.
53 Vec(int size) {
54 reserve(size);
55 this->size_ = size;
56 }
57
58 Vec(int size, T val) { resize(size, val); }
59
60 Vec(const Vec<T> &vec) {
61 this->size_ = vec.size();
62 this->capacity_ = this->size_;
63 auto policy = autoPolicy(this->size_);
64 if (this->size_ != 0) {
65 this->ptr_ = reinterpret_cast<T *>(malloc(this->size_ * sizeof(T)));
66 if (this->ptr_ == nullptr) throw std::bad_alloc();
67 TracyAllocS(this->ptr_, this->size_ * sizeof(T), 3);
68 uninitialized_copy(policy, vec.begin(), vec.end(), this->ptr_);
69 }
70 }
71
72 Vec(const std::vector<T> &vec) {
73 this->size_ = vec.size();
74 this->capacity_ = this->size_;
75 auto policy = autoPolicy(this->size_);
76 if (this->size_ != 0) {
77 this->ptr_ = reinterpret_cast<T *>(malloc(this->size_ * sizeof(T)));
78 if (this->ptr_ == nullptr) throw std::bad_alloc();
79 TracyAllocS(this->ptr_, this->size_ * sizeof(T), 3);
80 uninitialized_copy(policy, vec.begin(), vec.end(), this->ptr_);
81 }
82 }
83
84 Vec(Vec<T> &&vec) {
85 this->ptr_ = vec.ptr_;
86 this->size_ = vec.size_;
87 capacity_ = vec.capacity_;
88 vec.ptr_ = nullptr;
89 vec.size_ = 0;
90 vec.capacity_ = 0;
91 }
92
93 operator VecView<T>() { return {this->ptr_, this->size_}; }
94 operator VecView<const T>() const { return {this->ptr_, this->size_}; }
95
96 ~Vec() {
97 if (this->ptr_ != nullptr) {
98 TracyFreeS(this->ptr_, 3);
99 free(this->ptr_);
100 }
101 this->ptr_ = nullptr;
102 this->size_ = 0;
103 capacity_ = 0;
104 }

106 Vec<T> &operator=(const Vec<T> &other) {
107 if (&other == this) return *this;
108 if (this->ptr_ != nullptr) {
109 TracyFreeS(this->ptr_, 3);
110 free(this->ptr_);
111 }
112 this->size_ = other.size_;
113 capacity_ = other.size_;
114 auto policy = autoPolicy(this->size_);
115 if (this->size_ != 0) {
116 this->ptr_ = reinterpret_cast<T *>(malloc(this->size_ * sizeof(T)));
117 if (this->ptr_ == nullptr) throw std::bad_alloc();
118 TracyAllocS(this->ptr_, this->size_ * sizeof(T), 3);
119 uninitialized_copy(policy, other.begin(), other.end(), this->ptr_);
120 }
121 return *this;
122 }

147 inline void push_back(const T &val) {
148 if (this->size_ >= capacity_) {
149 // avoid dangling pointer in case val is a reference of our array
150 T val_copy = val;
151 reserve(capacity_ == 0 ? 128 : capacity_ * 2);
152 this->ptr_[this->size_++] = val_copy;
153 return;
154 }
155 this->ptr_[this->size_++] = val;
156 }
157
158 void reserve(int n) {
159 if (n > capacity_) {
160 T *newBuffer = reinterpret_cast<T *>(malloc(n * sizeof(T)));
161 if (newBuffer == nullptr) throw std::bad_alloc();
162 TracyAllocS(newBuffer, n * sizeof(T), 3);
163 if (this->size_ > 0)
164 uninitialized_copy(autoPolicy(this->size_), this->ptr_,
165 this->ptr_ + this->size_, newBuffer);
166 if (this->ptr_ != nullptr) {
167 TracyFreeS(this->ptr_, 3);
168 free(this->ptr_);
169 }
170 this->ptr_ = newBuffer;
171 capacity_ = n;
172 }
173 }
174
175 void resize(int newSize, T val = T()) {
176 bool shrink = this->size_ > 2 * newSize;
177 reserve(newSize);
178 if (this->size_ < newSize) {
179 uninitialized_fill(autoPolicy(newSize - this->size_),
180 this->ptr_ + this->size_, this->ptr_ + newSize, val);
181 }
182 this->size_ = newSize;
183 if (shrink) shrink_to_fit();
184 }
185
186 void shrink_to_fit() {
187 T *newBuffer = nullptr;
188 if (this->size_ > 0) {
189 newBuffer = reinterpret_cast<T *>(malloc(this->size_ * sizeof(T)));
190 if (newBuffer == nullptr) throw std::bad_alloc();
191 TracyAllocS(newBuffer, this->size_ * sizeof(T), 3);
192 uninitialized_copy(autoPolicy(this->size_), this->ptr_,
193 this->ptr_ + this->size_, newBuffer);
194 }
195 if (this->ptr_ != nullptr) {
196 TracyFreeS(this->ptr_, 3);
197 free(this->ptr_);
198 }
199 this->ptr_ = newBuffer;
200 capacity_ = this->size_;
201 }

203 VecView<T> view(int offset = 0, int length = -1) {
204 if (length == -1) {
205 length = this->size_ - offset;
206 if (length < 0) throw std::out_of_range("Vec::view out of range");
207 } else if (offset + length > this->size_ || offset < 0) {
208 throw std::out_of_range("Vec::view out of range");
209 } else if (length < 0) {
210 throw std::out_of_range("Vec::view negative length is not allowed");
211 }
212 return VecView<T>(this->ptr_ + offset, length);
213 }
214
215 VecView<const T> cview(int offset = 0, int length = -1) const {
216 if (length == -1) {
217 length = this->size_ - offset;
218 if (length < 0) throw std::out_of_range("Vec::cview out of range");
219 } else if (offset + length > this->size_ || offset < 0) {
220 throw std::out_of_range("Vec::cview out of range");
221 } else if (length < 0) {
222 throw std::out_of_range("Vec::cview negative length is not allowed");
223 }
224 return VecView<const T>(this->ptr_ + offset, length);
225 }
226
227 VecView<const T> view(int offset = 0, int length = -1) const {
228 return cview(offset, length);
229 }

231 T *data() { return this->ptr_; }
232 const T *data() const { return this->ptr_; }

234 private:
235 int capacity_ = 0;
236};
